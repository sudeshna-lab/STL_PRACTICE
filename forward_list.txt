#include <iostream>
#include <forward_list>

using namespace std;

void print(forward_list<int> f) {

    for(int value : f) {
        cout << value << " ";
    }

    cout << endl;
}

int main() {

    forward_list<int> f;

    // push_front()
    f.push_front(30);
    f.push_front(20);
    f.push_front(10);

    cout << "After push_front(): ";
    print(f);

    // front()
    cout << "Front element: " << f.front() << endl;

    // insert_after()
    auto it = f.begin();

    f.insert_after(it, 15);

    cout << "After insert_after(): ";
    print(f);

    // emplace_after()
    f.emplace_after(it, 12);

    cout << "After emplace_after(): ";
    print(f);

    // pop_front()
    f.pop_front();

    cout << "After pop_front(): ";
    print(f);

    // remove()
    f.remove(15);

    cout << "After remove(15): ";
    print(f);

    // reverse()
    f.reverse();

    cout << "After reverse(): ";
    print(f);

    // sort()
    f.sort();

    cout << "After sort(): ";
    print(f);

    // unique()
    f.push_front(12);
    f.push_front(12);

    cout << "Before unique(): ";
    print(f);

    f.unique();

    cout << "After unique(): ";
    print(f);

    // clear()
    f.clear();

    if(f.empty()) {
        cout << "List is empty" << endl;
    }

    return 0;
}
-------------------------------------------------------

Forward List Functions
Function	Description
push_front()	Insert at front
pop_front()	Remove first element
front()	Access first element
begin()	Iterator to beginning
end()	Iterator to end
before_begin()	Iterator before first element
empty()	Check if list is empty
max_size()	Maximum size
clear()	Remove all elements
insert_after()	Insert after position
emplace_after()	Construct element after position
erase_after()	Delete after position
remove()	Remove matching values
remove_if()	Remove by condition
sort()	Sort list
reverse()	Reverse list
unique()	Remove consecutive duplicates
merge()	Merge sorted lists
splice_after()	Transfer elements
resize()	Change size
assign()	Assign new values
swap()	Swap lists
emplace_front()	Construct at front
